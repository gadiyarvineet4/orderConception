    def setRequestAttributes(self, orderCreationMSG):
        self.orderNumber=orderCreationMSG["orderNumber"]
        self.countryCode=orderCreationMSG["countryCode"]
        self.seller=orderCreationMSG["seller"]
        self.orderType=orderCreationMSG["orderType"]
        self.orderDate=orderCreationMSG["orderDate"]
        requestShipToAddress=orderCreationMSG["shipToAddress"]
        self.shipToAddress.setAddressAttributes(requestShipToAddress)
        requestBillToAddress=orderCreationMSG["billToAddress"]
        self.billToAddress.setAddressAttributes(requestBillToAddress)
        requestCharges=orderCreationMSG["charges"]
        self.charges.setCharges(requestCharges)
        requestPaymentInfo=orderCreationMSG["paymentInfo"]
        self.paymentInfo.setPaymentInfo(requestPaymentInfo)
        requestWorkOrder=orderCreationMSG["workOrder"]
        self.workOrder.setWorkOrder(requestWorkOrder)
        requestItemLines=orderCreationMSG["itemLines"]
        
        for items in requestItemLines:
            requestItem = self.itemLines.setItemLines(items)
            self.itemLines.append(requestItem)

    def setItemLines(self,requestItemLines):
        self.itemType=requestItemLines["itemType"]
        self.itemId=requestItemLines["itemId"]
        self.itemName=requestItemLines["itemName"]
        self.quantity=requestItemLines["quantity"]
        self.sku=requestItemLines["sku"]
        self.description=requestItemLines["description"]
        self.ship_node=requestItemLines["ship_node"]
        return self

    def setWorkOrder(self, requestWorkOrder):
        self.workOrderNumber=requestWorkOrder["workOrderNumber"]
        self.workOrderSGR=requestWorkOrder["workOrderSGR"]
        self.appointmentStartTime=requestWorkOrder["appointmentStartTime"]
        self.appointmentEndTime=requestWorkOrder["appointmentEndTime"]
    
    def setPaymentInfo(self,requestPaymentInfo):
        self.status=requestPaymentInfo["status"]
        self.paymentType=requestPaymentInfo["paymentType"]

    def setAddressAttributes(self, requestAddress):
        self.customerFirstName=requestAddress["customerFirstName"]
        self.customerLastName=requestAddress["customerLastName"]
        self.customerPhoneNumber=requestAddress["customerPhoneNumber"]
        self.customerAddressLine1=requestAddress["customerAddressLine1"]
        self.customerAddressLine2=requestAddress["customerAddressLine2"]
        self.customerState=requestAddress["customerState"]
        self.customerZipCode=requestAddress["customerZipCode"]

    def setCharges(self,requestCharges):
        self.total=requestCharges["total"]
        self.itemTotal=requestCharges["itemTotal"]
        self.deliveryTotal=requestCharges["deliveryTotal"]
        self.tax=requestCharges["tax"]
